◆ startPauseResumeJobAction

final Action org.openpnp.gui.JobPanel.startPauseResumeJobAction
Initial value:
= new AbstractAction() {
{
putValue(SMALL_ICON, Icons.start);
putValue(NAME, Translations.getString("JobPanel.Action.Job.Start"));
putValue(SHORT_DESCRIPTION, Translations.getString("JobPanel.Action.Job.Start.Description"));
}
@Override
public void actionPerformed(ActionEvent arg0) {
UiUtils.messageBoxOnException(() -> {
if (state == State.Stopped) {
setState(State.Running);
jobStart();
}
else if (state == State.Paused) {
setState(State.Running);
jobRun();
}
else if (state == State.Running) {
setState(State.Pausing);
}
else {
throw new Exception("Don't know how to change from state " + state);
}
});
}
}
